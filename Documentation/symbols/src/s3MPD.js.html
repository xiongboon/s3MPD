1 /**
  2  * @constructor
  3  * @property {S3} s3 An Amazon S3 API instance
  4  * @property {integer} currentFile Keeps track of the current file number being downloaded
  5  * @property {integer} fullSize Keeps track of the full download size for the array
  6  * @property {array} fileSizes Array of individual file sizes
  7  * @property {array} fileArray Array of files to download
  8  * @property {string} bucket Amazon S3 bucket
  9  * @property {array} chunkDownloadProgress Keeps track of inidividual file/chunk download progress
 10  * @property {function} fileSavedFunction Function that gets called when a file is saved

 12  * @property {integer} fileSize File size of the currently downloading file

 14  * @property {string} objectName Keeps track of the file name of the file currently being downloaded
 15  * @property {integer} tries Keeps track of the number of retry attempts for a single file
 16  * @property {integer} range How many bytes a single chunk should be
 17  * @property {integer} i_max Total number of chunks to break the file into
 18  * @property {uint8array} tempBlob Temporary blob to hold individual chunks that get downloaded
 19  * @property {array} allChunks Temporary array to keep track of which chunks have already been downloaded
 20  * @property {array} chunkStaging Temporary array to keep track of how many chunks are currently downloading
 21  * @property {uint8array} fullBlob Completed blob of downloaded file
 22  * @property {integer} chunkSize Size of each chunk to download
 23  * @property {integer} chunks Number of chunks to download at the same time
 24  * @property {integer} currentChunk Keeps track of the current chunk being downloaded
 25  * @property {string} localMD5 MD5 of the current downloaded file
 26  * @property {integer} averageSpeed Average download speed
 27  * @property {integer} lastSpeed Last recorded download speed
 28  * @property {function} callback Callback function (should remove this later and figure out a better way to pass the callback to the setTimeout)
 29  * @property {boolean} gettingSize Designates whether or not the progress bar should show that it is getting the full download size
 30  * @property {integer} sizeDone Number of files that have successfully gotten their download size
 31  * @param {string} keyID Amazon IAM Key ID
 32  * @param {string} accessKey Amazon IAM Access Key
 33  * @param {string} region Amazon S3 Region
 34  * @param {function} callback Callback function
 35  */
 36 function s3MPD(keyID, accessKey, region, callback) {
 37 	AWS.config.update({accessKeyId: keyID, secretAccessKey: accessKey});
 38 	AWS.config.region = region;
 39 
 40 	this.s3 = new AWS.S3();
 41 
 42 	this.getInitialDownloadSpeed(function(speed) {
 43 		callback();
 44 	});
 45 }

 58 /**
 59  * Downloads an array of files
 60  * @param {object} params Download parameters
 61  * @param {function} callback Callback function
 62  * @param {function} fileSavedFunction Function that gets called when a file is saved
 63  * @param {function} updateProgressBarFunction Function that gets called to update a progress bar
 64  * @config {array} [files] Array of files to download
 65  * @config {string} [bucket] Amazon S3 bucket
 66  */
 67 s3MPD.prototype.downloadArray = function(params, callback, fileSavedFunction, updateProgressBarFunction) {
 68 	var _this = this;
 69 
 70 	_this.diagLog('Downloading Array');
 71 
 72 	_this.currentFile = 0;
 73 	_this.fullSize = 0;
 74 
 75 	_this.fileSizes = new Array();
 76 
 77 	_this.fileArray = params.files;
 78 	_this.bucket = params.bucket;
 79 
 80 	_this.chunkDownloadProgress = new Array();
 81 
 82 	if (fileSavedFunction) {
 83 		_this.fileSavedFunction = fileSavedFunction;
 84 	} else {
 85 		_this.fileSavedFunction = null;
 86 	}
 87 
 88 	if (updateProgressBarFunction) {
 89 		_this.updateProgressBarFunction = updateProgressBarFunction;
 90 	} else {
 91 		_this.updateProgressBarFunction = null;
 92 	}
 93 
 94 	/** Internal function to start download process */
 95 	function internal_download() {
 96 		_this.fileSize = _this.fileSizes[_this.currentFile];
 97 
 98 		if (_this.currentFile < _this.fileArray.length) {
 99 			_this.does_file_exist(_this.fileArray[_this.currentFile], function(bool) {
100 				if (bool) {
101 					_this.diagLog('File Already Exists: ' + _this.fileArray[_this.currentFile]);
102 
103 					_this.getLocalFileSize(_this.fileArray[_this.currentFile], function(size) {
104 						_this.chunkDownloadProgress[_this.currentFile] = new Array();
105 						_this.chunkDownloadProgress[_this.currentFile][0] = size;
106 
107 						_this.getFileURL(_this.fileArray[_this.currentFile], function(fileURL) {
108 							if (_this.fileSavedFunction) {
109 								_this.fileSavedFunction(_this.currentFile, _this.fileArray[_this.currentFile], fileURL);
110 							}
111 
112 							_this.currentFile++;
113 
114 							internal_download();
115 						});
116 					});
117 				} else {
118 					_this.diagLog('Downloading File ' + (_this.currentFile + 1) + ' / ' + (_this.fileArray.length + 1));
119 
120 					_this.download(_this.fileArray[_this.currentFile], function() {
121 						_this.saveCurrentFile(function(success) {
122 							if (success) {
123 								_this.currentFile++;
124 
125 								internal_download();
126 							}
127 						});
128 					});
129 				}
130 			});
131 		} else {
132 			callback();
133 		} 
134 	}
135 
136 	_this.getFullDownloadSize(function() {
137 		internal_download();
138 	});
139 }

243 /**
244  * Downloads a single file
245  * @param {string} object File name to download
246  * @param {function} callback Callback function
247  */
248 s3MPD.prototype.download = function(object, callback) {
249 	var _this = this;
250 
251 	_this.objectName = object;
252 	_this.tries = 0;
253 
254 	_this.chunkDownloadProgress[_this.currentFile] = new Array();
255 
256 	_this.getParameters(function() {
257 		_this.diagLog('Starting Download...');
258 	
259 		var fileSize = _this.fileSize;
260 		var range = _this.chunkSize;
261 
262 		if ( range >= fileSize ) {
263 			range = fileSize;
264 		}
265 
266 		_this.range = range;
267 
268 		var i_max = Math.ceil(fileSize / _this.range);
269 
270 		_this.i_max = i_max;
271 
272 		if ( _this.chunks > _this.i_max ) {
273 			_this.chunks = _this.i_max;
274 		}
275 
276 		_this.tempBlob = new Uint8Array( fileSize );
277 		_this.allChunks = new Array();
278 		_this.chunkStaging = new Array();
279 
280 		for (var x = 0; x < _this.i_max; x++) {
281 			_this.allChunks.push(x);
282 		}
283 
284 		_this.chunkListener(callback);
285 	});
286 }

288 /**
289  * Stages chunks to be downloaded and waits for all chunks to finish downloading
290  * @param {function} callback Callback function
291  */
292 s3MPD.prototype.chunkListener = function(callback) {
293 	var _this = this;
294 
295 	setTimeout(function() {
296 		if (_this.chunkStaging.length < _this.chunks){
297 			var nextChunk = _this.allChunks.pop();
298 
299 			if (nextChunk != undefined) {
300 				_this.chunkStaging.push(true);
301 
302 				_this.downloadChunk(nextChunk, function (blob, y) {
303 					if (blob) {
304 						_this.tempBlob.set( new Uint8Array( blob ), y * _this.range );
305 						_this.chunkStaging.pop();
306 
307 						_this.chunkListener(callback);
308 					} else {
309 						_this.diagLog('File Has Been Skipped');
310 					}
311 				});
312 
313 				_this.chunkListener(callback);
314 			} else {
315 				if ((_this.chunkStaging.length == 0) && (_this.tempBlob != null)) {
316 					_this.diagLog('All Chunks Downloaded!');
317 
318 					_this.fullBlob = new Blob([_this.tempBlob.buffer], {type: _this.contentType });
319 					_this.tempBlob = null;
320 
321 					_this.getBlobMD5(function() {
322 						_this.remoteMD5 = _this.eTag;
323 
324 						if (_this.localMD5 == _this.remoteMD5) {
325 							_this.diagLog(_this.objectName + ': MD5 Success');
326 							
327 							callback();
328 					    } else if (_this.remoteMD5.indexOf('-')) {
329 					    	_this.diagLog(_this.objectName + ': Not able to compare MD5 Hashes (multipart upload)');
330 					    	
331 					    	callback();
332 					    } else {
333 					    	_this.diagLog(_this.objectName + ': MD5 Failure');
334 
335 					    	_this.fullBlob = null;
336 					    	_this.download(_this.objectName, callback);
337 					    }
338 					});
339 				} else {
340 					setTimeout(function() {
341 						_this.chunkListener(callback);
342 					}, 100, callback, _this);
343 				}
344 			}
345 		}
346 	}, 100, callback, _this);
347 }

349 /**
350  * Gets download parameters based on initial download speed
351  * @param {function} callback Callback function
352  */
353 s3MPD.prototype.getParameters = function(callback) {
354 	var _this = this;
355 
356 	_this.diagLog('Getting Download Parameters');
357 
358 	var speed = _this.averageSpeed;
359 	var fileSize = _this.fileSize;
360 
361 	if (speed == 0) {
362 		var maxChunkSize = 102400; //100KB
363 	} else {
364 		var maxChunkSize = Math.ceil((fileSize / Math.ceil((fileSize / speed) / 120)) * .8);
365 	}
366 
367 	if (speed <= 512000) {
368 		_this.chunkSize = Math.min(maxChunkSize, 1048576);
369 
370 		_this.chunks = 1;
371 	} else {
372 		switch (true) {
373 			case (speed < 1048576): //1MB per second
374 				_this.chunkSize = 5 * 1024 * 1024; //5MB
375 
376 				_this.chunks = 3;
377 
378 				break;	
379 			case (speed < 3932160): //3.75MB per second
380 				_this.chunkSize = 10 * 1024 * 1024; //10MB
381 
382 				_this.chunks = 4;
383 
384 				break;
385 			default: //anything over 3.75MB per second
386 				_this.chunkSize = 15 * 1024 * 1024; //15MB
387 
388 				_this.chunks = 4;
389 
390 				break;
391 		}
392 	}
393 
394 	callback();
395 }

397 /**
398  * Updates the average download speed
399  * @param {function} callback Callback function
400  */
401 s3MPD.prototype.updateAverageSpeed = function(callback) {
402 	var _this = this;
403 
404 	var smoothingFactor = .005;
405 	var lastSpeed = _this.lastSpeed;
406 	var averageSpeed = _this.averageSpeed;
407 
408 	_this.averageSpeed = (smoothingFactor * lastSpeed) + ((1 - smoothingFactor) * averageSpeed); //Exponential moving average algorithm
409 
410 	callback();
411 }

413 /**
414  * Downloads an individual chunk
415  * @param {integer} i Chunk that should be downloaded
416  * @param {function} callback Callback function
417  */
418 s3MPD.prototype.downloadChunk = function(i, callback) {
419 	var _this = this;
420 
421 	var fileSize = _this.fileSize;
422 
423 	_this.chunkDownloadProgress[_this.currentFile][i] = 0; 
424 	_this.currentChunk = i;
425 
426 	var next = (i + 1);
427 
428 	if ( next >= _this.i_max) {
429 		var rng = 'bytes='+ (_this.range * i) + '-' +  fileSize;
430 	} else {
431 		var rng = 'bytes='+ (_this.range * i) + '-' + ((_this.range * i) + (_this.range - 1));
432 	}
433 
434 	_this.diagLog('Downloading Chunk ' + (i+1) + "/" + _this.i_max);
435 
436 	var params = {
437 		Bucket: _this.bucket,
438 		Key: _this.objectName,
439 		Range: rng
440 	};
441 	
442 	var req = _this.s3.getObject( params );
443 	
444 	req.on('error', function(msg){
445 		_this.tries++;
446 
447 		if (_this.tries <= 5) {
448 			_this.diagLog('Chunk ' + i + ' failed. Attempt #' + _this.tries + '. Retrying...');
449 
450 			_this.downloadChunk(i, callback);
451 		} else {
452 			_this.diagLog('Chunk ' + i + ' failed. Attempt #' + _this.tries + '. Skipping File...');
453 
454 			callback(false);
455 		}
456 	});
457 
458 	req.on('httpError', function(error, response) {
459 		_this.tries++;
460 
461 		if (_this.tries <= 5) {
462 			_this.diagLog('Chunk ' + i + ' failed. Attempt #' + _this.tries + '. Retrying...');
463 
464 			_this.downloadChunk(i, callback);
465 		} else {
466 			_this.diagLog('Chunk ' + i + ' failed. Attempt #' + _this.tries + '. Skipping File...');
467 
468 			callback(false);
469 		}
470 	});
471 
472 	req.on('httpDownloadProgress', function (evt) {
473 		_this.chunkDownloadProgress[_this.currentFile][i] = evt.loaded;
474 		
475 		_this.diagLog("Download Progress: " + _this.objectName + " | " + (i + 1) + " / " + _this.i_max +" - " +  (parseInt(evt.loaded) / parseInt(evt.total) * 100).toFixed(2) + "%");
476 
477 		_this.updateProgressBarFunction();
478 	});
479 
480 	req.on('complete', function (response) {
481 		if (response.error) {
482 			_this.diagLog('Chunk ' + i + ' failed. Retrying...');
483 
484 			_this.downloadChunk(i, callback);
485 		} else {
486 			_this.diagLog('Finished Chunk ' + (i+1) + "/" + _this.i_max);
487 
488 			var end = new Date();
489 			end = end.getTime();
490 
491 			var time = ((end - start) / 1000);
492 			var speed = fileSize / time;
493 
494 			_this.lastSpeed = speed;
495 
496 			_this.updateAverageSpeed(function() {
497 				_this.contentType = response.data.ContentType;
498 				_this.eTag = response.data.ETag.substring(1, response.data.ETag.length-1);
499 
500 				callback(response.data.Body, i);
501 			});
502 		}
503 	});
504 
505 	var start = new Date();
506 	start = start.getTime();
507 
508 	req.send();
509 }

529 /**
530  * Gets the initial download speed
531  * @param {function} callback Callback function
532  */
533 s3MPD.prototype.getInitialDownloadSpeed = function(callback) {
534 	var _this = this;
535 
536 	_this.diagLog('Getting Initial Download Speed');
537 
538 	var fileSize = 1000000;
539 	var totalBytes = 0;
540 
541 	if (_this.tries == null) {
542 		_this.tries = 0;
543 	}
544 
545 	var params = {
546 		Bucket: 'torokiosk',
547 		Key: '1meg.test'
548 	};
549 	
550 	var req = _this.s3.getObject( params );
551 	
552 	req.on('error', function(msg){
553 		if (_this.tries > 5) {
554 			req.abort();
555 
556 			req = null;
557 
558 			_this.diagLog('Cannot Get Initial Download Speed');
559 
560 			callback(false);
561 		}
562 
563 		_this.tries++;
564 
565 		_this.getInitialDownloadSpeed(callback);
566 	});
567 
568 	req.on('httpDownloadProgress', function (progress, response) {
569 		if (progress.loaded < totalBytes) {
570 			var end = new Date();
571 			end = end.getTime();
572 
573 			var time = ((end - start) / 1000);
574 			var speed = fileSize / time;
575 
576 			req.abort();
577 
578 			_this.tries = null;
579 			req = null;
580 
581 			_this.diagLog('Initial Download Speed: ' + (speed / 1024 / 1024) + 'MB/s');
582 
583 			callback(speed);
584 		}
585 
586 		totalBytes = progress.loaded;
587 	});
588 
589 	req.on('complete', function (response) {
590 		if (response.error) {
591 			if (_this.tries > 5) {
592 				req.abort();
593 
594 				req = null;
595 
596 				_this.diagLog('Cannot Get Initial Download Speed');
597 
598 				callback(false);
599 			}
600 
601 			_this.tries++;
602 
603 			_this.getInitialDownloadSpeed(callback);
604 		} else {
605 			var end = new Date();
606 			end = end.getTime();
607 
608 			var time = ((end - start) / 1000);
609 			var speed = fileSize / time;
610 
611 			_this.averageSpeed = speed;
612 			_this.lastSpeed = speed;
613 
614 			_this.diagLog('Initial Download Speed: ' + (speed / 1024 / 1024) + 'MB/s');
615 
616 			callback(speed);
617 		}
618 	});
619 
620 	var start = new Date();
621 	start = start.getTime();
622 
623 	req.send();
624 }

626 /**
627  * Gets the full download size for the array
628  * @param {function} callback Callback function
629  */

631 	var _this = this;
632 
633 	_this.callback = callback;
634 	_this.gettingSize = true;
635 
636 	_this.diagLog('Getting Full Download Size');
637

639 
640 	_this.fileArray.forEach(function(value, key) {
641 		_this.diagLog('Getting File Size: ' + (key + 1) + ' / ' + _this.fileArray.length);
642 
643 		var params = {
644 			Bucket: _this.bucket,
645 			Key: value
646 		};
647 
648 		_this.s3.headObject(params, function(err, data) {
649 			if (err) {
650 				_this.diagLog(err + ' ' + err.stack);
651 			} else {
652 				if (data.ContentLength == undefined) {
653 					var xhr = $.ajax({
654 						type: "HEAD",
655 						url: ('https://s3.amazonaws.com/torokiosk/' + value),
656 						success: function(msg) {
657 							_this.fullSize += parseInt(xhr.getResponseHeader('Content-Length'));
658 							_this.fileSizes[key] = parseInt(xhr.getResponseHeader('Content-Length'));

661

663 						}
664 					});	
665 				} else {
666 					_this.fullSize += parseInt(data.ContentLength);
667 					_this.fileSizes[key] = parseInt(data.ContentLength);
668 
669 					_this.updateProgressBarFunction();
670 
671 					_this.sizeDone++;
672 				}
673 		  	}
674 		});
675 	});
